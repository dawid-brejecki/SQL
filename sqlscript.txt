employees ;-- podwyzka dla pracownikow o 10%, ale dla Kinga o 50%select employee_id, salary * (case when last_name = 'King' then 1.5 else 1.1 end) as nowa_pensja from employees;-- PIVOTselect * from (select job_id, department_id, count(*) as liczba from employees group by job_id, department_id) pivot  (sum(liczba) for department_id in (80,90,100));-- informacja o pracownikach i liczbie osob posiadajˆcych taka sama liczbe znakow w emailu jak oni w danych dzialachselect employee_id, first_name, last_name, email, length(email),  count(*) over (partition by department_id, length(email)) from employees;-- informacje o stanowiskach, wraz ze skumulowana maksymalna pensja, posortowanych wg job_idselect job_id, job_title, max_salary, sum(max_salary) over (order by job_id rows unbounded preceding) from jobs ;-- lista wysokosci zarobkow w danym dziale i danym managerzeselect department_id, manager_id, listagg(salary, ',') within group (order by department_id, manager_id) as lista from employees group by department_id, manager_id ;-- lista pracownikow zarabiajacych najlepiej w swoich dzia¸ach wraz z informacja o id pracownika zarabiajˆcych najlepiej zaraz po nichselect employee_id, first_name, last_name, salary, department_id, ono from ( select employee_id, first_name, last_name, salary, department_id, row_number() over (partition by department_id order by salary desc) as rn, lead(employee_id) over (partition by department_id order by salary desc) as ono from employees ) where rn = 1;-- lista pracownikow najwczesniej zatrudnionych w swoich dzialachselect * from ( select first_name, department_id, hire_date,  dense_rank() over (partition by department_id order by hire_date) as liczba from employees) where liczba = 1;-- wyswietlenie listy pracownikow zatrudnionych w 2005 roku oraz listy pracownik—w pracujˆcych w ITselect first_name, last_name, hire_date, job_id from employees where to_char(hire_date,'YYYY') in (2005) UNION ALL select first_name, last_name, hire_date, job_id from employees  where job_id like 'IT_PROG';-- wyswietlenie listy pracownikow zatrudnionych w 2005 roku i pracujˆcych w ITselect first_name, last_name, hire_date, job_id from employees where to_char(hire_date,'YYYY') in (2005) INTERSECT select first_name, last_name, hire_date, job_id from employees  where job_id like 'IT_PROG' ;-- wyswietlenie listy pracownikow zatrudnionych w 2005 roku, ale nie pracujˆcych w ITselect first_name, last_name, hire_date, job_id from employees where to_char(hire_date,'YYYY') in (2005) MINUS select first_name, last_name, hire_date, job_id from employees  where job_id like 'IT_PROG';-- wyswietlenie informacji o pracownikach, ich nazwy stanowiska i nazwy departamentu, nie uwgledniajac pracownik—w z ITselect e.employee_id, e.first_name, e.salary, j.job_title, d.department_name from employees e join jobs j on e.job_id = j.job_id join departments d on e.department_id = d.department_id where d.department_name != 'IT';
